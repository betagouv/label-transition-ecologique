import {Meta, ColorPalette, ColorItem} from '@storybook/blocks';
import {preset} from './tailwind-preset.ts';

<Meta title="Colors" />

<ColorPalette>
{
    Object.entries(preset.theme.extend.colors).map(([key, value]) => {
        if (typeof value === 'object') {
            const {DEFAULT, ...others} = value;
            if (!DEFAULT) {
                return <ColorItem title={key} colors={value} />
            }

            return (<>
                <ColorItem title={`${key} (DEFAULT)`} colors={{DEFAULT}} />
                <ColorItem title={key} colors={{...others}} />
            </>)
        }

       return (
            <ColorItem title={key} colors={{[key]: value}} />
        )
    })

}
</ColorPalette>
